# ```mermaid
graph TD
    %% Уровень 1: Начало
    Start([НАЧАЛО]) --> Q1{Бюджет?}
    
    %% Уровень 2: Ветвление по бюджету
    Q1 -->|< 800 тыс| ClassA[Класс A]
    Q1 -->|800к-1.5м| ClassB[Класс B]
    Q1 -->|> 1.5м| Q2{Цель?}
    
    %% Уровень 3: Детализация по бюджету
    ClassA --> Q2
    
    ClassB --> Q2
    
    Q2 -->|Город| Q3{Тип кузова?}
    Q2 -->|Семья| Univ[Универсал/Минивэн]
    Q2 -->|Бездорожье| AWD[Полный привод]
    
    %% Уровень 4: Тип кузова
    Q3 -->|Седан| Sedan[Седан]
    Q3 -->|Хэтчбек| Hatch[Хэтчбек]
    
    Sedan --> Q4{КПП?}
    Hatch --> Q4
    Univ --> Q4
    AWD --> Q4
    
    %% Уровень 5: Коробка передач
    Q4 -->|Автомат| AT[Автомат]
    Q4 -->|Механика| MT[Механика]
    
    AT --> Q5{Приоритет?}
    MT --> Q5
    
    %% Уровень 6: Приоритеты
    Q5 -->|Экономичность| Eco[Эконом. двигатель]
    Q5 -->|Безопасность| Safe[Высокий рейтинг]
    Q5 -->|Комфорт| Comfort[Доп. опции]
    
    %% Уровень 7: Поиск и результат
    Eco --> Search[ПОИСК В БД]
    Safe --> Search
    Comfort --> Search
    
    Search --> Found{Найдено?}
    
    Found -->|Да| Rank[Ранжирование]
    Found -->|Нет| Expand[Расширить параметры]
    
    Expand --> Search
    
    Rank --> Top3[ТОП-3 рекомендации]
    Top3 --> Warn[Предупреждения]
    Warn --> Result([РЕЗУЛЬТАТ])
    
    %% Стили для наглядности
    style Start fill:#2980b9,color:#fff
    style Result fill:#27ae60,color:#fff
    style Q1,Q2,Q3,Q4,Q5 fill:#f39c12
    style Found fill:#e74c3c
    style Search fill:#3498db,color:#fff
```
